#include<stdio.h>
#include<stdlib.h>
#include<pic18f4450.h>

void main(void)
{
    int i,j,temp;
    int num[] ={10,2,5,1,16};
    
    for(i=0;i<=4;i++)
    {
        for(j=i+1;j<=4;j++)
        {
            if(num[i]>num[j])
            {
                temp = num[i];
                num[i] = num[j];
                num[j] = temp;
            }
        }
    }
            
            

}
